# 初始Vue

**主要是了解一下Vue在干啥子**



**说明：将hello world!作用在id在root下**

```vue
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Hello world</title>
    <script src="https://unpkg.com/vue@next"></script>
</head>
<body>
    <div id="root"></div>
</body>

<script>
    Vue.createApp({
        template:'<div>hello world!</div>'
    }).mount('#root');
</script>
</html>
```



**说明：变量**

```vue
<script>
    Vue.createApp({
        data(){
            return{
                content:1
            }
        },
        template:'<div>{{content}}</div>'
    }).mount('#root');
</script>
</html>
```



**说明：mounted()页面加载完成后自动执行的函数**

```vue
<script>
    Vue.createApp({
        data(){
            return{
                content:1
            }
        },
        // 类似析构函数
        mounted(){
            setInterval(()=>{
                // this指向data.content
                this.content += 1;
            },1000)
        },
        template:'<div>{{content}}</div>'
    }).mount('#root');
</script>
</html>
```



**说明:v-on:click="需要绑定的函数名称"**

```vue
<script>
    Vue.createApp({
        data() {
            return {
                content: 'hello world'
            }
        },
        methods:{
            handleBtnClick(){
                // 反转
                this.content = this.content.split('').reverse().join('');
            }
        },
        template:
            `<div>
                {{content}}
                <button v-on:click="handleBtnClick">反转</div>
            </div>`
    }).mount('#root');
</script>
</html>
```



**说明  v-if="条件" 隐藏与展示**

```vue
<!-- 格式化代码 ALT+SHIFT+F -->
<script>
    Vue.createApp({
        data() {
            return {
                show :true
            }
        },
        methods:{
            handleBtnClick(){
                // 反转
                this.show = !this.show
            }
        },
        template:
            `<div>
                <spand v-if="show">Hello world</spand>
                <button v-on:click="handleBtnClick">展示/隐藏</div>
            </div>`
    }).mount('#root');
</script>
```



**说明：数据的双向绑定** {{}}-》插值表达式

```vue
<!-- 格式化代码 ALT+SHIFT+F -->
<script>
    Vue.createApp({
        data() {
            return {
                inputValue:'',
                list:[],//数组
            }
        },
        methods:{
            handleAddItem(){
                this.list.push(this.inputValue);
                this.inputValue = ''
            }
        },
        template:
            `<div>
                <!-- 数据双向绑定 -->
                <input v-model="inputValue"/>
                <button v-on:click="handleAddItem">增加</button>
                <ul>
                    <!-- 对数组内容循环 -->
                    <li v-for="(item,index) of list">{{item}} {{index}}</li>
                </ul>
            </div>`
    }).mount('#root');
</script>
</html>
```



**说明：v-bind: 绑定标签的属性**

        data() {

**说明：拆分组件和简单的使用**

```vue
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Hello world</title>
    <script src="https://unpkg.com/vue@next"></script>
</head>



<body>
    <div id="root"></div>
</body>

<!-- 格式化代码 ALT+SHIFT+F -->
<script>
    const app = Vue.createApp({
        data() {
            return {
                inputValue:'',
                list:[],//数组
            }
        },
        methods:{
            handleAddItem(){
                this.list.push(this.inputValue);
                this.inputValue = ''
            }
        },
        template:
            `<div>
                <!-- 数据双向绑定 -->
                <input v-model="inputValue"/>
                <button v-on:click="handleAddItem"  v-bind:title="inputValue">增加</button>
                <ul>
                    <!-- 对数组内容循环 -->
                    <!-- 数据绑定 -->
                    <todo-item
                     v-for="(item,index) of list"
                     v-bind:content="item"
                     v-bind:index="index"
                     />
                </ul> 
            </div>`
    });


    // 定义小组件
    app.component('todo-item',{
        // 接收数据
        props:['content','index'],
        template:'<li>{{index}} -- {{content}}</li>'
    });

    // 作用与id = root
    app.mount('#root');
</script>
</html>
```
